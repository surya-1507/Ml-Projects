def text_transform(text):
    text=text.lower()
    text=nltk.word_tokenize(text)
    temp=[]
    for i in text:
        if i.isalnum():
            temp.append(i)
    text=temp[:]
    temp=[]
    for i in text:
        if i not in stopwords.words('english') and i not in string.punctuation:
            temp.append(i)
    text=temp[:]
    temp=[]
    for i in text:
        temp.append(ps.stem(i))
        
    return ' '.join(temp)
